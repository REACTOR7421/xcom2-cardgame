const express = require('express'); const path = require('path'); const app = express(); const PORT = process.env.PORT || 10000; // Statikus fájlok kiszolgálása app.use(express.static(path.join(__dirname, 'public'))); app.get('/', (req, res) => { res.sendFile(path.join(__dirname, 'public', 'index.html')); }); app.listen(PORT, () => { console.log(`Szerver fut: http://localhost:${PORT}`); });
